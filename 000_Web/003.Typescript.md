## Interface[optional,readonly]
- 인터페이스는 객체를 위한 타입을 지정할 때 사용되는 문법이다.
```ts
type Score  = `A`|`B`|`C`|`D`|

interface User{
    name:string;
    age: number;
    gender?:string;
    readonly birthYear:number;
    [grade:number]:Score,
}

let user:User{
    name:"Elizabeth",
    age:30,
    birthYear:2000,
    grade:`A`
}

user.name = "Che Rin"
user.age = 10
user.gender = "female"

```

## Interface Class 

```ts
interface Car{
    color:string;
    wheels:number;
    start():void;
}

interface Car2 extends Car{
    door:number;
    stop():void;
}

interface Car3 extends Car,Car2{
    price:number;
}

class BMW implements Car{
    color; 
    wheels = 4; 
    constructor(c:string){
        this.color = c;
    }
    start({
        console.log("go");
    })
}

const b = new BMW("green")
```
